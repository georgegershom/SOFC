PYTHON USAGE
------------

from material_properties_loader import YSZMaterialProperties

# Initialize
ysz = YSZMaterialProperties()

# Get property at temperature
E_800 = ysz.get_property('Youngs_Modulus_GPa', 800)
print(f"E at 800°C: {E_800:.2f} GPa")  # Output: 118.00 GPa

# Get all properties
props = ysz.get_all_properties(800)

# Calculate creep rate
creep = ysz.get_creep_rate(stress_mpa=50, temperature_c=1000, grain_size_um=1.5)

# Plot properties
ysz.plot_properties(save_path='properties.png')

# Export for FEM
import numpy as np
temps = np.linspace(25, 1200, 50)
ysz.export_for_fem('fem_input.csv', temperature_points=temps)
